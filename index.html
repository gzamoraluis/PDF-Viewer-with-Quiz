// Quiz configuration const QUIZ_TITLE = "Document Knowledge Assessment"; const PASSING_SCORE = 7; // 70% to pass // Quiz questions - replace with your actual questions const quizQuestions = [ { question: "What is the main purpose of this document?", options: ["To provide guidelines", "To share research findings", "To announce an event", "To sell a product"], answer: 0 }, { question: "Which section contains the summary of key points?", options: ["Introduction", "Methodology", "Executive Summary", "Appendix"], answer: 2 }, // Add more questions... ]; // Initialize the quiz form function initQuiz() { const quizForm = document.getElementById('quiz-form'); quizForm.innerHTML = ''; quizQuestions.forEach((q, i) => { const questionDiv = document.createElement('div'); questionDiv.className = 'question'; const questionText = document.createElement('p'); questionText.textContent = `${i+1}. ${q.question}`; questionDiv.appendChild(questionText); q.options.forEach((option, j) => { const input = document.createElement('input'); input.type = 'radio'; input.name = `question-${i}`; input.value = j; input.id = `q${i}-opt${j}`; input.required = true; const label = document.createElement('label'); label.htmlFor = `q${i}-opt${j}`; label.textContent = option; questionDiv.appendChild(input); questionDiv.appendChild(label); questionDiv.appendChild(document.createElement('br')); }); quizForm.appendChild(questionDiv); }); } // Calculate quiz score and gather answers function calculateResults() { let score = 0; const userAnswers = []; const currentDate = new Date(); quizQuestions.forEach((q, i) => { const selectedOption = document.querySelector(`input[name="question-${i}"]:checked`); const answerIndex = selectedOption ? parseInt(selectedOption.value) : null; const isCorrect = answerIndex === q.answer; if (isCorrect) score++; userAnswers.push({ question: q.question, options: q.options, userAnswer: answerIndex !== null ? q.options[answerIndex] : "Not answered", correctAnswer: q.options[q.answer], isCorrect: isCorrect }); }); return { score: score, maxScore: quizQuestions.length, percentage: Math.round((score / quizQuestions.length) * 100), passed: score >= PASSING_SCORE, date: currentDate.toLocaleDateString(), time: currentDate.toLocaleTimeString(), answers: userAnswers }; } // Generate the PDF report function generatePDFReport(results) { const doc = new jsPDF(); // Add title doc.setFontSize(18); doc.setTextColor(40, 53, 147); doc.text(QUIZ_TITLE, 105, 20, { align: 'center' }); // Add date and time doc.setFontSize(12); doc.setTextColor(0, 0, 0); doc.text(`Date: ${results.date}`, 20, 35); doc.text(`Time: ${results.time}`, 20, 45); // Add score summary doc.setFontSize(14); doc.setTextColor(results.passed ? 46, 125, 50 : 198, 40, 40); doc.text(`Score: ${results.score}/${results.maxScore} (${results.percentage}%)`, 105, 60, { align: 'center' }); doc.text(`Result: ${results.passed ? 'PASSED' : 'FAILED'}`, 105, 70, { align: 'center' }); // Add questions and answers doc.setFontSize(12); doc.setTextColor(0, 0, 0); let yPosition = 85; results.answers.forEach((answer, index) => { if (yPosition > 270) { doc.addPage(); yPosition = 20; } doc.setFont(undefined, 'bold'); doc.text(`${index + 1}. ${answer.question}`, 20, yPosition); yPosition += 8; doc.setFont(undefined, 'normal'); doc.setTextColor(answer.isCorrect ? 46, 125, 50 : 198, 40, 40); doc.text(`✓ Your answer: ${answer.userAnswer}`, 25, yPosition); yPosition += 8; if (!answer.isCorrect) { doc.setTextColor(46, 125, 50); doc.text(`✓ Correct answer: ${answer.correctAnswer}`, 25, yPosition); yPosition += 8; } doc.setTextColor(0, 0, 0); yPosition += 10; }); // Save the PDF const fileName = `Quiz_Report_${results.date.replace(/\//g, '-')}.pdf`; doc.save(fileName); } // Display the quiz results function displayResults(results) { const reportInfo = document.getElementById('report-info'); const reportQuestions = document.getElementById('report-questions'); // Display summary reportInfo.innerHTML = `

Date: ${results.date}

Time: ${results.time}

Score: ${results.score}/${results.maxScore} (${results.percentage}%) - ${results.passed ? 'PASSED' : 'FAILED'}

`; // Display all questions and answers reportQuestions.innerHTML = '

Question Details

'; results.answers.forEach((answer, i) => { const answerDiv = document.createElement('div'); answerDiv.className = 'question'; answerDiv.innerHTML = `

Question ${i + 1}: ${answer.question}

Your answer: ${answer.userAnswer}

${!answer.isCorrect ? `

Correct answer: ${answer.correctAnswer}

` : ''} `; reportQuestions.appendChild(answerDiv); }); // Show report section document.getElementById('quiz-section').style.display = 'none'; document.getElementById('report-section').style.display = 'block'; } // Event listeners document.getElementById('open-pdf').addEventListener('click', function() { // Open PDF from repository window.open(PDF_PATH, '_blank'); // Show quiz button document.getElementById('take-quiz-btn').style.display = 'inline-block'; }); document.getElementById('take-quiz-btn').addEventListener('click', function() { // Hide PDF section, show quiz document.getElementById('pdf-section').style.display = 'none'; document.getElementById('quiz-section').style.display = 'block'; initQuiz(); }); document.getElementById('submit-quiz').addEventListener('click', function(e) { e.preventDefault(); // Check all questions answered const allAnswered = quizQuestions.every((_, i) => { return document.querySelector(`input[name="question-${i}"]:checked`); }); if (!allAnswered) { alert('Please answer all questions before submitting.'); return; } // Calculate and display results const results = calculateResults(); displayResults(results); }); document.getElementById('retry-quiz').addEventListener('click', function() { // Return to PDF section document.getElementById('quiz-section').style.display = 'none';
